135  def control_modality(): 
136      """Cross validation of Random10K against Medline100K results 
137      in multiple modes in the article score distributions. 
138       
139      The distribution becomes unimodal about zero if we eliminate features that 
140      do not occur in positive citations. There are thousands of such features 
141      all of which have a negative feature scores of roughly the same value (-9). 
142      Random occurrences of 0,1,2 or 3 of those -9 features in test citations 
143      causes the score of that citation to be shifted by -9, -18, -27 from 
144      zero.""" 
145      rc.get_postmask = "mask_nonpositives" 
146      pos, neg = dataset_map["control"] 
147      op = CrossValidation(rc.working / "valid" / "modality", 
148                           "Removing Control corpus multi-modality")
