 30      @param results: Result PubMed IDs in decreasing order of score 
 31       
 32      @param test: Set of known relevant articles (to be counted as true positives)

 34      @return: Vector of true positives at each rank (false

 36      assert isinstance(test, set) 
 37      TP_total = nx.zeros(len(results), nx.int32) 
 38      for idx, pmid in enumerate(results): 
 39          TP_total[idx] = TP_total[idx-1] if idx > 0 else 0 
 40          if pmid in test: 
 41              TP_total[idx] += 1
