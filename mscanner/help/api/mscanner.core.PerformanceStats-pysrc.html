248          rule. 
249           
250          @note: The vectors r1 .. r7 correspond to rows of Table II in 
251          Hanley1982. 
252   
253          @return: L{W}, L{W_stderr} 
254          """ 
255          s = self 
256          # r1 is number of negatives with each score, 
257          # r2 is number of positives rated higher than each score 
258          # r3 is number of positives with each score 
259          # r4 is number of negatives rated lower than each score 
260          r1 = s.NE 
261          r2 = s.TP - s.PE 
262          r3 = s.PE 
263          r4 = s.TN 
264          r5 = r1 * r2 + 0.5 * r1 * r3 
265          r6 = r3 * (r4**2 + r4*r1 + (r1**2)/3) 
266          r7 = r1 * (r2**2 + r2*r3 + (r3**2)/3) 
267          N = float(s.N) 
268          P = float(s.P) 
269          W = r5.sum() / (N*P) 
270          Q2 = r6.sum() / (P * N**2) 
271          Q1 = r7.sum() / (N * P**2) 
272          W_stderr = nx.sqrt((W*(1-W)+(P-1)*(Q1-W**2)+(N-1)*(Q2-W**2))/(P*N)) 
273          #print W, Q1, Q2, W_stderr
